Piccolo script di awk che calcola media e varianza. La media è giusta, e anche la varianza dovrebbe
awk '{sum+=$4;sumsq+=$4*$4;}
	{avg=sum/NR;} 
	END{print "avg =",avg,$5;
		var=sumsq/NR-(avg*avg); 
		print "var =",var,$5; 
		stdev=sqrt(var); 
		print "stdev =",stdev;
		}' ../Prove\ sparse/Prove\ files\ di\ testo/prova1.txt
